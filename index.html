// المُعرّف الخاص بملفك var SHEET_ID = "12fvogtYxUKiXnrPbY_YmFL-2S7wTqXUkxxOh_FLFJV0"; // دالة لتهيئة الجداول وتنسيقها تلقائياً function setupDatabase() { var ss = SpreadsheetApp.openById(SHEET_ID); // 1. إعداد وتنسيق جدول الطلبات var reqSheet = ss.getSheetByName("Requests"); if (!reqSheet) { reqSheet = ss.insertSheet("Requests"); var headers = ["معرف الطلب", "التاريخ", "اسم العميل", "رقم الجوال", "الخدمة", "الوصف", "المرفق", "الحالة", "اسم الفني", "وقت البدء", "وقت الانتهاء"]; reqSheet.appendRow(headers); var range = reqSheet.getRange(1, 1, 1, headers.length); range.setFontWeight("bold").setBackground("#1e3a8a").setFontColor("white").setHorizontalAlignment("center"); reqSheet.setFrozenRows(1); } // 2. إعداد وتنسيق جدول المنتجات var prodSheet = ss.getSheetByName("Products"); if (!prodSheet) { prodSheet = ss.insertSheet("Products"); var pHeaders = ["معرف المنتج", "اسم الخدمة / المنتج", "السعر"]; prodSheet.appendRow(pHeaders); var pRange = prodSheet.getRange(1, 1, 1, pHeaders.length); pRange.setFontWeight("bold").setBackground("#f59e0b").setFontColor("white").setHorizontalAlignment("center"); prodSheet.setFrozenRows(1); // إضافة خدمات افتراضية prodSheet.appendRow([Date.now(), "سباكة", "مشمول بالباقة"]); prodSheet.appendRow([Date.now()+1, "كهرباء", "مشمول بالباقة"]); } return { reqSheet: reqSheet, prodSheet: prodSheet }; } // دالة لجلب البيانات وعرضها لجميع الأجهزة (المزامنة) function doGet(e) { if (e.parameter.action === "getData") { var db = setupDatabase(); var pRows = db.prodSheet.getDataRange().getValues(); var rRows = db.reqSheet.getDataRange().getValues(); var products = []; for (var i = 1; i < pRows.length; i++) products.push({ name: pRows[i][1], price: pRows[i][2] }); var requests = []; for (var j = 1; j < rRows.length; j++) { requests.push({ id: rRows[j][0], date: rRows[j][1], user: rRows[j][2], phone: rRows[j][3], service: rRows[j][4], desc: rRows[j][5], image: rRows[j][6], status: rRows[j][7], tech: rRows[j][8], start: rRows[j][9], end: rRows[j][10] }); } return ContentService.createTextOutput(JSON.stringify({ products: products, requests: requests })) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput("✅ السكربت وقاعدة البيانات تعمل باحترافية وبدون أخطاء!"); } // دالة استقبال الإضافات والتعديلات function doPost(e) { try { var db = setupDatabase(); var data = JSON.parse(e.postData.contents); var action = data.action; // إضافة طلب جديد if (action === "addRequest") { var imageUrl = "بدون مرفق"; if (data.imageBase64 && data.imageBase64 !== "") { var folders = DriveApp.getFoldersByName("صور_طلبات_الصيانة"); var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder("صور_طلبات_الصيانة"); var blob = Utilities.newBlob(Utilities.base64Decode(data.imageBase64), data.imageMimeType, "image.jpg"); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); imageUrl = file.getUrl(); } db.reqSheet.appendRow([ data.id, data.date, data.user, data.phone, data.service, data.desc, imageUrl, "قيد الانتظار", "لم يحدد", "لم يبدأ", "لم ينتهِ" ]); return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON); } // تحديث حالة الطلب والأوقات والفني if (action === "updateStatus") { var rows = db.reqSheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0].toString() === data.id.toString()) { db.reqSheet.getRange(i + 1, 8).setValue(data.status); // تحديث الحالة if (data.tech) db.reqSheet.getRange(i + 1, 9).setValue(data.tech); // اسم الفني if (data.start) db.reqSheet.getRange(i + 1, 10).setValue(data.start); // وقت البدء if (data.end) db.reqSheet.getRange(i + 1, 11).setValue(data.end); // وقت الانتهاء break; } } return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON); } // إضافة منتج جديد من الإدارة if (action === "addProduct") { db.prodSheet.appendRow([data.id, data.name, data.price]); return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON); } } catch(err) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()})).setMimeType(ContentService.MimeType.JSON); } }
